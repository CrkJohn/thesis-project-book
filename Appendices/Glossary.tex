\newcommand{\glossarydef}[4]{
    \newglossaryentry{#1}{
        text={#2},
        long={#3},
        name={{\sl \glsentrylong{#1} (\glsentrytext{#1}\/})},
        first={\glsentryname{#1}},
        firstplural={\glsentrylong{#1}\glspluralsuffix (\glsentryname{#1}\glspluralsuffix )},
        description={{\rm #4}}
    }
}

\newcommand{\simpleglossarydef}[3]{
    \newglossaryentry{#1}{
        name={{\rm \sl #2}},
        description={{\rm #3}}
    }
}
